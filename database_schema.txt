Table element_type {
  id integer [pk, increment]
  type varchar 
}

Table rarity {
  id integer [pk, increment]
  name varchar
}

Table user {
  id          integer [pk, increment]
  email       varchar [unique]
  username    varchar [unique]
  password    varchar [not null]
}

Table character {
  id integer [pk, increment]
  name varchar
  description varchar
  rarity integer [ref: > rarity.id ]
  element_type integer [ref: > element_type.id ]
}

Table user_own_characters {
  id integer [pk, increment]
  user_id integer [not null, ref: <> user.id ]
  character_id integer [not null, ref: <> character.id ]
  souls_amount integer [not null, default: 0]
  character_level integer [not null, default: 1]
}
